## 一、快速开始

### 1.1 安装

```bash
# 安装依赖
pip install -r requirements.txt

# 安装为开发模式
pip install -e .

# 初始化配置
cognix init
```

### 1.2 基本使用

```python
from cognix.interfaces.agent import AgentMemoryClient

# 创建记忆客户端
memory = AgentMemoryClient()

# 添加记忆
memory.remember("user_name", {"name": "张三", "department": "研发部"}, "fact")

# 查询记忆
result = memory.recall(key="user_name")
print(result.data)

# 搜索记忆
result = memory.recall(query="研发")
print(result.data)

# 删除记忆
memory.forget("user_name")
```

## 二、核心功能使用

### 2.1 添加记忆

#### 添加事实记忆

```python
memory = AgentMemoryClient()

# 添加用户信息
result = memory.remember(
    key="user_zhangsan",
    value={"name": "张三", "email": "zhangsan@example.com", "department": "研发部"},
    memory_type="fact",
    confidence=1.0
)

print(f"记忆ID: {result.data['memory_id']}")
```

#### 添加偏好记忆

```python
# 添加输出格式偏好
memory.remember(
    key="report_format",
    value={"value": "table", "last_used": "2024-05-20"},
    memory_type="preference"
)

# 添加时间偏好
memory.remember(
    key="work_time",
    value={"start": "09:00", "end": "18:00", "lunch_time": "12:00-13:00"},
    memory_type="preference"
)
```

#### 添加经验记忆

```python
# 添加工具使用经验
memory.remember(
    key="tool_search_usage",
    value={
        "tool_name": "web_search",
        "usage_pattern": ["查询天气", "查找资料", "获取新闻"],
        "success_rate": 0.95,
        "best_time": "工作日 9:00-17:00"
    },
    memory_type="experience",
    confidence=0.9
)
```

---

### 2.2 查询记忆

#### 精确查询

```python
result = memory.recall(key="user_zhangsan")
if result.success:
    print(result.data)
else:
    print(f"查询失败: {result.message}")
```

#### 模糊搜索

```python
# 搜索包含"张三"的记忆
result = memory.recall(query="张三")
for item in result.data:
    print(f"- {item['key']}: {item['value']}")

# 按类型搜索
result = memory.recall(query="report")
for item in result.data:
    print(f"- {item['memory_type']}: {item['key']}")
```

---

### 2.3 删除记忆

```python
result = memory.forget("user_zhangsan")
if result.success:
    print("删除成功")
else:
    print(f"删除失败: {result.message}")
```

### 2.4 获取上下文

```python
# 获取会话上下文（用于Agent推理）
session_id = "user_session_123"
result = memory.contextualize(session_id, query="周报")

if result.success:
    context = result.data
    print("短期记忆:", context["short_term"])
    print("事实记忆:", context["long_term"]["facts"])
    print("偏好记忆:", context["long_term"]["preferences"])
    print("经验记忆:", context["long_term"]["experiences"])
```

---

### 2.5 记录交互

```python
# 记录Agent与用户的交互
session_id = "user_session_123"
memory.learn(
    session_id=session_id,
    user_input="帮我生成周报",
    agent_response="已生成本周周报，格式为表格",
    tool_calls=[
        {"tool": "generate_report", "args": {"format": "table"}}
    ]
)
```

---

### 2.6 建立关联

```python
# 建立记忆之间的关联
memory.remember("project_x", {"name": "项目X", "status": "进行中"}, "fact")
memory.remember("task_1", {"name": "任务1", "project": "项目X"}, "fact")

# 建立关联关系
memory.connect("project_x", "task_1", relation="related_to")

# 查询关联记忆
result = memory.get_connections("project_x")
for item in result.data:
    print(f"关联记忆: {item['key']}")
```

## 三、与 OpenClaw Agent 集成

### 3.1 集成示例

```python
from cognix.interfaces.agent import AgentMemoryClient

class OpenClawAgent:
    def __init__(self):
        self.memory = AgentMemoryClient()
        self.session_id = None
    
    def set_session(self, session_id):
        self.session_id = session_id
    
    def process_message(self, user_input):
        # 1. 获取上下文记忆
        context_result = self.memory.contextualize(self.session_id, user_input)
        context = context_result.data if context_result.success else {}
        
        # 2. 构建提示词
        prompt = self._build_prompt(user_input, context)
        
        # 3. 调用LLM
        response = self._call_llm(prompt)
        
        # 4. 记录交互
        self.memory.learn(
            session_id=self.session_id,
            user_input=user_input,
            agent_response=response
        )
        
        # 5. 更新长期记忆（根据响应内容）
        self._update_memories(response)
        
        return response
```

### 3.2 工作流程

```
┌─────────────────────────────────────────────────────────────┐
│                     OpenClaw Agent                         │
├─────────────────────────────────────────────────────────────┤
│  用户输入                                                   │
│      │                                                     │
│      ▼                                                     │
│  contextualize(session_id, query)                          │
│      │                                                     │
│      ▼                                                     │
│  获取短期记忆 + 长期记忆                                     │
│      │                                                     │
│      ▼                                                     │
│  注入Prompt → LLM推理                                       │
│      │                                                     │
│      ▼                                                     │
│  learn(session_id, input, response)                         │
│      │                                                     │
│      ▼                                                     │
│  记录到短期记忆                                             │
│      │                                                     │
│      ▼                                                     │
│  定期沉淀到长期记忆                                          │
└─────────────────────────────────────────────────────────────┘
```

## 四、CLI 命令

### 4.1 查看记忆

```bash
# 查看所有长期记忆
cognix memory list

# 按类型查看
cognix memory list --type fact
cognix memory list --type preference
cognix memory list --type experience

# 搜索记忆
cognix memory search "关键词"
```

### 4.2 添加记忆

```bash
# 添加事实记忆
cognix memory add --type fact --key "user_name" --value '{"name": "张三"}'

# 添加偏好记忆
cognix memory add --type preference --key "report_format" --value '{"value": "table"}'
```

### 4.3 删除记忆

```bash
cognix memory delete user_name
```

### 4.4 查看记忆关联

```bash
cognix memory relations user_name
```

## 八、注意事项

1. **记忆键唯一性**：每个记忆的 key 必须唯一，重复 key 会覆盖现有记忆
2. **置信度范围**：置信度取值范围为 0-1，值越高表示记忆越可靠
3. **短期记忆过期**：短期记忆默认 24 小时过期，可通过配置调整
4. **数据备份**：建议定期备份 `~/.cognix/` 目录
5. **性能考虑**：大量记忆时建议使用 search API 的 limit 参数限制返回数量
